  <script>

    const gameState = {

      player: {

        name: "Người chơi 1",

        job: "Kỹ sư",

        salary: 1500,

        expenses: 1000,

        assets: [],

        liabilities: [],

        cash: 2000,

        passiveIncome: 0,

        netWorth: 0,

        risk: 0.1

      },

      economy: {

        fedRate: 5.25,

        unemployment: 3.8,

        pmi: 51.3,

        inflation: 3.5,

        news: ["Fed nâng lãi suất thêm 0.25%", "PMI dịch vụ tăng nhẹ", "Tổng thống ban hành đạo luật hỗ trợ startup"]

      },

      opportunities: [

        { type: "real_estate", name: "Mua căn hộ cho thuê", price: 60000, income: 500, risk: 0.05 },

        { type: "stock", name: "Cổ phiếu AI startup", price: 5000, income: 0, chance: 0.3, risk: 0.2 },

        { type: "crypto", name: "Đầu tư Bitcoin", price: 3000, income: 0, chance: 0.5, risk: 0.4 },

        { type: "business", name: "Startup ứng dụng học AI", price: 10000, income: 1000, risk: 0.3 }

      ],

      events: [

        { text: "Khủng hoảng tài chính - thị trường giảm 30%", effect: (state) => {

          state.player.cash *= 0.7;

          state.player.risk += 0.1;

        }},

        { text: "Chính phủ hỗ trợ người dân - nhận thêm 2000$", effect: (state) => {

          state.player.cash += 2000;

        }}

      ]

    };


    function updateUI() {

      const p = gameState.player;

      document.getElementById('playerInfo').innerHTML = `

        <h3>Thông tin người chơi</h3>

        <p><strong>Nghề nghiệp:</strong> ${p.job}</p>

        <p><strong>Tiền mặt:</strong> $${p.cash.toFixed(2)}</p>

        <p><strong>Thu nhập bị động:</strong> $${p.passiveIncome}</p>

        <p><strong>Chi phí:</strong> $${p.expenses}</p>

        <p><strong>Giá trị ròng:</strong> $${calculateNetWorth(p)}</p>

        <p><strong>Mức rủi ro:</strong> ${(p.risk * 100).toFixed(1)}%</p>

      `;


      const oppDiv = document.getElementById('opportunities');

      oppDiv.innerHTML = '';

      gameState.opportunities.forEach((opp, idx) => {

        const btnDisabled = p.cash < opp.price ? 'disabled' : '';

        const item = document.createElement('div');

        item.className = 'col';

        item.innerHTML = `

          <div style="padding:10px; background:#e2e8f0; border-radius:8px">

            <strong>${opp.name}</strong><br>

            💵 Giá: $${opp.price}<br>

            📈 Thu nhập: $${opp.income || 0}<br>

            ⚠️ Rủi ro: ${(opp.risk * 100).toFixed(1)}%<br>

            <button class="btn" ${btnDisabled} onclick="invest(${idx})">Đầu tư</button>

          </div>

        `;

        oppDiv.appendChild(item);

      });


      const newsUl = document.getElementById('newsList');

      newsUl.innerHTML = '';

      gameState.economy.news.forEach(news => {

        const li = document.createElement('li');

        li.innerText = news;

        newsUl.appendChild(li);

      });

    }


    function calculateNetWorth(player) {

      const assets = player.assets.reduce((sum, a) => sum + (a.value || 0), 0);

      const liabilities = player.liabilities.reduce((sum, l) => sum + (l.amount || 0), 0);

      return assets - liabilities + player.cash;

    }


    function invest(index) {

      const opp = gameState.opportunities[index];

      const p = gameState.player;

      if (p.cash >= opp.price) {

        p.cash -= opp.price;

        p.assets.push({ ...opp, value: opp.price });

        if (opp.income) p.passiveIncome += opp.income;

        if (opp.risk) p.risk += opp.risk;

        updateUI();

      }

    }


    function simulateMonth() {

      const p = gameState.player;

      p.cash += p.salary + p.passiveIncome - p.expenses;

      p.netWorth = calculateNetWorth(p);

      if (Math.random() < p.risk) {

        const event = gameState.events[Math.floor(Math.random() * gameState.events.length)];

        event.effect(gameState);

        alert("[TIN KHẨN] " + event.text);

      }

      updateUI();

    }


    updateUI();
